# Dharani - Pendency Code

## Total Pendency

```
var pendency = 0
var totalApplications = 0


tableBody = document.querySelector('#a > div > div > table > tbody')


rows = tableBody.querySelectorAll('tr')


rows.forEach(row => {
    let td = row.querySelectorAll('td')
    let total = parseInt(td[3].innerHTML)
    let disposed = parseInt(td[5].innerHTML)
    pendency += total - disposed
    totalApplications += total
})


percentagePending = parseFloat((pendency * 100 / totalApplications).toFixed(2))
var today = new Date(Date.now())
var todayString = `${today.getDate()}/${today.getMonth() + 1}/${today.getFullYear()}`
const data = []
data.push({
    date:todayString,
    application: totalApplications,
    pendency: pendency,
    percentage: percentagePending
})


console.table(data)
```
