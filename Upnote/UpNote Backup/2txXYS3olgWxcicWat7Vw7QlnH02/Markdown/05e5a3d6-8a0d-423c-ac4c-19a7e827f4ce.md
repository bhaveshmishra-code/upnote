# Amazon Order Code

  

```
dateList = []
priceList = []
orderList = []


spans = document.querySelectorAll('span.value')
spans.forEach((span,i) => {
    if(i%3 === 0){
        dateList.push(span.innerText)
    }else if(i%3 == 1){
        priceList.push(parseFloat(span.innerText.replace(',','')))
    }else{
        orderList.push(span.innerText)
    }
})


const orders = []
for(let i=0; i < dateList.length; i++){
    console.log(`${dateList[i]},${orderList[i]},${priceList[i]}`)
    orders.push({
        id: orderList[i],
        date: dateList[i],
        price: priceList[i]
    })
}
console.table(orders)
```
